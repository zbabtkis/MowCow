/**
 * A base game entity to render in a world
 *
 * @class Entity
 */
function Entity(location, options) {
  this.id = _.uniqueId('entity_');
  this.options = options;
  this.x = location.x;
  this.y = location.y;
  this._onMoveNotifiers = [];
}

/**
 * Render the entity in the context of the world
 *
 * @function render
 * @memberof Entity#
 * @return {Object} X and Y pixel ranges entity exists in
 * @property {Array<Integer>} x X pixel range
 * @property {Array<Integer>} y Y pixel range
 */
Entity.prototype.render = function() {
  var ctx = this.world.ctx;

  ctx.beginPath();
  ctx.lineWidth = "0";
  ctx.fillStyle = this.options.color;
  ctx.fillRect(this.x, this.y, this.options.width, this.options.height);
  ctx.stroke();

  return {
    x: _.range(this.x, this.x + this.options.width),
    y: _.range(this.y, this.y + this.options.height)
  };
};

/**
 * Teardown entity
 *
 * @function destroy
 * @memberof Entity#
 * @return {Entity} Instance of this entity
 */
Entity.prototype.destroy = function() {
  this._onMoveNotifiers = [];
  return this;
};

/**
 * Register a function to call when the entity moves
 *
 * @function onMove
 * @memberof Entity#
 * @param {Function} notifier Function to call when movement occurs
 * @return {Entity} Instance of this entity
 */
Entity.prototype.onMove = function(notifier) {
  this._onMoveNotifiers.push(notifier);
};

module.exports = Entity;
